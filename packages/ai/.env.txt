import type {
  ProductionPlan,
} from "../types/workflow";

import type {
  AIProvider,
} from "@skyra/ai/provider";

export class AIDirector {
  constructor(
    private ai: AIProvider
  ) {}

  async createPlan(
    userRequest: string
  ): Promise<ProductionPlan> {
    const systemPrompt = `
You are SKYRA AI Director.

You are the central intelligence
responsible for planning professional
AI-generated video productions.

You understand:

- Storytelling
- Cinematography
- Video generation
- Human avatars
- Character consistency
- Voice generation
- Lip synchronization
- Music
- Sound design
- Camera direction
- Lighting
- Visual continuity
- Aspect ratios

Transform the user's request into
a complete SKYRA production plan.

Return valid JSON only.
`;

    const response =
      await this.ai.generate(
        [
          {
            role:
              "system",

            content:
              systemPrompt,
          },

          {
            role:
              "user",

            content:
              userRequest,
          },
        ],

        {
          model:
            process.env
              .SKYRA_AI_MODEL ??
            "development",

          temperature:
            0.2,

          responseFormat:
            "json",
        }
      );

    return JSON.parse(
      response
    ) as ProductionPlan;
  }
}
